METHOD_BODY::= STATEMENT_LIST
STATEMENT_LIST::= STATEMENT STATEMENT_LIST_DASH
STATEMENT_LIST_DASH::= STATEMENT STATEMENT_LIST_DASH | ε
STATEMENT::= DECLARATION | IF | WHILE | ASSIGNMENT
DECLARATION::= PRIMITIVE_TYPE id ;
PRIMITIVE_TYPE::= int | float
IF::= if ( EXPRESSION ) { STATEMENT } else { STATEMENT }
WHILE::= while ( EXPRESSION ) { STATEMENT }
ASSIGNMENT::= id assign EXPRESSION ;
EXPRESSION::= SIMPLE_EXPRESSION TEMP_EXPRESSION
TEMP_EXPRESSION::= relop SIMPLE_EXPRESSION | ε
SIMPLE_EXPRESSION::= TEMP_SIMPLE_EXPRESSION SIMPLE_EXPRESSION_DASH
TEMP_SIMPLE_EXPRESSION::= TERM | SIGN TERM
SIMPLE_EXPRESSION_DASH::= addop TERM SIMPLE_EXPRESSION_DASH | ε
TERM::= FACTOR TERM_DASH
TERM_DASH::= mulop FACTOR TERM_DASH | ε
FACTOR::= id | num | ( EXPRESSION )
SIGN::= addop
